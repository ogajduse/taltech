no shutdown

ip address 192.168.2.1 255.255.255.0
no shut

no shut
exit


int s0/0/1

no shut
exit

network 10.1.1.0 0.0.0.255 area 0
network 10.2.2.0 0.0.0.255 area 0
network 192.168.12.0 0.0.0.255 area 0
network 192.168.2.0 0.0.0.255 area 0
exit

line vty 0 4
login local
transport input ssh
exit

crypto key generate rsa general-keys modulus 1024
username cisco algorithm-type scrypt secret cisco12345

int g0/1
ip access-group 102 in
exit

!!! access lists !!!
no ip access-list extended ACL_LAN
no ip access-list extended ACL_DMZ
no ip access-list extended ACL_TO_R3
no ip access-list extended ACL_TO_R1

deny object-group SSH object-group ALL-R2-INT object-group ALL-R2-INT log
permit object-group SSH object-group LAN object-group LAN log
permit object-group TCP-AND-UDP object-group LAN object-group DMZ log
permit object-group TCP-AND-UDP object-group LAN object-group DMZ log
exit

ip access-list extended ACL_DMZ
deny group SSH object-group ALL-R2-INT object-group ALL-R2-INT log
permit object-group SSH object-group LAN object-group LAN log
exit

ip access-list extended ACL_TO_R3
deny object-group SSH object-group ALL-R2-INT object-group ALL-R2-INT log
permit object-group SSH object-group LAN object-group LAN log
permit object-group OSPF object-group TO-R3 any log
exit

permit object-group SSH object-group LAN object-group LAN log
exit
